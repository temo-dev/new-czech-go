05 — Data Model & API

Exercise payload — cteni_6 / poslech_6

JSON schema
{
  "exercise_type": "cteni_6",          // or "poslech_6"
  "title":    "Úřední hodiny — Vlašim",
  "passage":  "Vlašim\nMěstský úřad...", // display text (cteni_6)
                                          // or TTS script (poslech_6)
  "questions": [
    { "statement": "Na úřadu města je zavřeno ve středu." },
    { "statement": "Ve čtvrtek je polední přestávka do jedné hodiny." },
    { "statement": "V úterý úřední hodiny končí ve dvě hodiny odpoledne." }
  ],
  "correct_answers": ["ne", "ne", "ano"], // parallel array
  "max_points": 3
}

// Submit (reuse existing endpoint):
POST /v1/attempts/:id/submit-answers
{ "answers": ["ne", "ano", "ano"] }

{
  "question_text":       "Ve čtvrtek je polední přestávka do jedné hodiny.",
  "learner_answer_text": "ANO",
  "correct_answer_text": "NE",
  "is_correct":          false
}
